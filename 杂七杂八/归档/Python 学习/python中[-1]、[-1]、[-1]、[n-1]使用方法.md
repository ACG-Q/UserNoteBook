```python
import numpy as np

a=np.random.rand(5)
print(a) # 输出 [0.26625424 0.97458008 0.34658445 0.34065681 0.93659931]

#取最后一个元素
print(a[-1]) #输出 0.9365993083726595

# 除了最后一个取全部
print(a[:-1])  #输出 [0.26625424 0.97458008 0.34658445 0.34065681]

# 取从后向前（相反）的元素
print(a[::-1]) # 输出 [0.93659931 0.34065681 0.34658445 0.97458008 0.26625424]

# 取从下标为2的元素翻转读取
print(a[2::-1]) #输出 [0.34658445 0.97458008 0.26625424]
```
